Random Forest Regressor Robust Evaluation & Model Diagnostics

This section trains a Random Forest on the leak-free train/test Parquets and visualizes:

  • Prediction quality: actual vs predicted, residuals vs predicted
  • Error by decile (calibration-ish check for trees)
  • Stability by year (test MAE grouped by fyear)
  • Feature importance (impurity) and Permutation Importance (on test)
  • Partial Dependence for top features (global effect, tree-friendly)

# compute permutation importance on TEST to avoid bias

    rf, X_test, y_test,
    n_repeats=10, random_state=42, n_jobs=-1

# Try new-style PDP API; fall back gracefully if unavailable
try:
    from sklearn.inspection import PartialDependenceDisplay as PDPDisp
    _HAS_PDP = True
except Exception:
    _HAS_PDP = False

if _HAS_PDP:
    import matplotlib.pyplot as plt
    for feat in top_feats:
        fig, ax = plt.subplots(figsize=(7,4.5))
        PDPDisp.from_estimator(rf, X_test, [feat], ax=ax, kind="average")
        ax.set_title(f"Partial Dependence  {feat}")
        plt.show()
else:
    # Fallback to old API if needed
    from sklearn.inspection import partial_dependence
    import numpy as np
    import matplotlib.pyplot as plt

    for feat in top_feats:
        j = FEATURE_COLS.index(feat)
        pd_result = partial_dependence(rf, X_test, [j])
        xs = pd_result["values"][0]
        avg = pd_result["average"][0]
        fig, ax = plt.subplots(figsize=(7,4.5))
        ax.plot(xs, avg)
        ax.set_xlabel(feat)
        ax.set_ylabel("Average predicted E/P")
        ax.set_title(f"Partial Dependence  {feat}")
        ax.grid(True, alpha=0.3)
        plt.show()
